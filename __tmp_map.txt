import React, { useEffect, useState } from "react";
import {
  View,
  Text,
  Button,
  StyleSheet,
  Modal,
  TextInput,
  TouchableOpacity,
  Alert,
  Switch,
  Image,
} from "react-native";
import MapView, { Marker, MapPressEvent, Callout } from "react-native-maps";
import * as Location from "expo-location";
import * as ImagePicker from "expo-image-picker";
import { signOut } from "firebase/auth";
import { collection, addDoc, getDocs , where, or , query } from "firebase/firestore";
import { ref, uploadBytes, getDownloadURL } from "firebase/storage";
import { auth, db, storage } from "../../firebaseConfig";
import { useRouter, useLocalSearchParams } from "expo-router";

export default function MapScreen() {
  const [region, setRegion] = useState<any>(null);
  const [markers, setMarkers] = useState<any[]>([]);
  const [modalVisible, setModalVisible] = useState(false);
  const [selectedCoord, setSelectedCoord] = useState<any>(null);
  const [name, setName] = useState("");
  const [desc, setDesc] = useState("");
  const [image, setImage] = useState<string | null>(null);
  const [isPublic, setIsPublic] = useState(false);
  const [detailItem, setDetailItem] = useState<any | null>(null);
  const router = useRouter();
  const params = useLocalSearchParams<{ latitude?: string; longitude?: string; name?: string; imageUrl?: string; description?: string }>();

  // üîπ R√©cup√®re la position actuelle ou celle d'une adresse cliqu√©e
  useEffect(() => {
    (async () => {
      const lat = params?.latitude ? parseFloat(params.latitude as string) : undefined;
      const lon = params?.longitude ? parseFloat(params.longitude as string) : undefined;

      if (typeof lat === "number" && !Number.isNaN(lat) && typeof lon === "number" && !Number.isNaN(lon)) {
        setRegion({ latitude: lat, longitude: lon, latitudeDelta: 0.01, longitudeDelta: 0.01 });
        return;
      }

      let { status } = await Location.requestForegroundPermissionsAsync();
      if (status !== "granted") {
        Alert.alert("Permission refus√©e", "L‚Äôacc√®s √† la localisation est n√©cessaire.");
        return;
      }

      const location = await Location.getCurrentPositionAsync({});
      setRegion({
        latitude: location.coords.latitude,
        longitude: location.coords.longitude,
        latitudeDelta: 0.01,
        longitudeDelta: 0.01,
      });
    })();

    loadAddresses();
  }, [params?.latitude, params?.longitude]);

// üîπ Charge uniquement les adresses visibles pour l'utilisateur

const loadAddresses = async () => {
  try {
    const currentUser = auth.currentUser?.email?.toLowerCase();

    const addressesRef = collection(db, "addresses");

    // 1) Publiques
    const publicSnap = await getDocs(query(addressesRef, where("isPublic", "==", true)));
    const publicDataRaw = publicSnap.docs.map(d => ({ id: d.id, ...d.data() } as any));

    // 2) Priv√©es de l'utilisateur connect√© (si connect√©)
    let privateDataRaw: any[] = [];
    if (currentUser) {
      const privateSnap = await getDocs(query(addressesRef, where("user", "==", currentUser)));
      privateDataRaw = privateSnap.docs.map(d => ({ id: d.id, ...d.data() } as any));
    }

    // 3) Fusion sans doublons
    const combinedRaw = [
      ...publicDataRaw,
      ...privateDataRaw.filter(p => !publicDataRaw.find(pub => pub.id === p.id)),
    ];

    // 4) Normaliser les donn√©es
    const combined = combinedRaw
      .map((m: any) => ({
        ...m,
        latitude: typeof m.latitude === 'string' ? Number(m.latitude) : m.latitude,
        longitude: typeof m.longitude === 'string' ? Number(m.longitude) : m.longitude,
        isPublic: !!m.isPublic,
        user: (m.user ? String(m.user) : '').toLowerCase(),
      }))
      .filter((m: any) => typeof m.latitude === 'number' && !Number.isNaN(m.latitude) && typeof m.longitude === 'number' && !Number.isNaN(m.longitude));

    setMarkers(combined);
  } catch (error) {
    console.error("Erreur lors du chargement :", error);
  }
};


  // üîπ Clique sur la carte ‚Üí pr√©pare ajout d'adresse
  const handleAddMarker = (e: MapPressEvent) => {
    setSelectedCoord(e.nativeEvent.coordinate);
    setModalVisible(true);
  };

  // Appui long: afficher les d√©tails du marqueur le plus proche (si suffisamment proche)
  const handleLongPress = (e: MapPressEvent) => {
    const { latitude, longitude } = e.nativeEvent.coordinate;
    if (!markers || markers.length === 0) return;
    const toRad = (deg: number) => (deg * Math.PI) / 180;
    const R = 6371000; // Terre en m√®tres
    const distance = (m: any) => {
      const lat1 = toRad(latitude);
      const lat2 = toRad(Number(m.latitude));
      const dLat = toRad(Number(m.latitude) - latitude);
      const dLon = toRad(Number(m.longitude) - longitude);
      const a = Math.sin(dLat / 2) ** 2 + Math.cos(lat1) * Math.cos(lat2) * Math.sin(dLon / 2) ** 2;
      return 2 * R * Math.asin(Math.sqrt(a));
    };
    let nearest = markers[0];
    let best = distance(nearest);
    for (let i = 1; i < markers.length; i++) {
      const d = distance(markers[i]);
      if (d < best) {
        best = d;
        nearest = markers[i];
      }
    }
    if (best <= 120) {
      setDetailItem(nearest);
    }
  };

  // üîπ S√©lection d‚Äôimage
  const pickImage = async () => {
    let result = await ImagePicker.launchImageLibraryAsync({
      mediaTypes: ImagePicker.MediaTypeOptions.Images,
      allowsEditing: true,
      quality: 0.7,
    });

    if (!result.canceled) {
      setImage(result.assets[0].uri);
    }
  };

  // üîπ Sauvegarde l‚Äôadresse + image
  // const saveAddress = async () => {
  //   if (!name.trim()) {
  //     Alert.alert("Erreur", "Veuillez entrer un nom d‚Äôadresse.");
  //     return;
  //   }

  //   let imageUrl = "";

  //   try {
  //     if (image) {
  //       const response = await fetch(image);
  //       const blob = await response.blob();
  //       const filename = `images/${Date.now()}_${auth.currentUser?.uid}.jpg`;
  //       const storageRef = ref(storage, filename);
  //       await uploadBytes(storageRef, blob);
  //       imageUrl = await getDownloadURL(storageRef);
  //     }

  //    await addDoc(collection(db, "addresses"), {
  //     name,
  //     description: desc,
  //     latitude: selectedCoord.latitude,
  //     longitude: selectedCoord.longitude,
  //     imageUrl,
  //     // üëá Toujours un email propre en minuscules
  //     user: (auth.currentUser?.email || "inconnu").toLowerCase(),
  //     // üëá Toujours un bool√©en vrai/faux
  //     isPublic: !!isPublic,
  //     createdAt: new Date(),
  //   });


  //     Alert.alert("Succ√®s", "Adresse ajout√©e !");
  //     setModalVisible(false);
  //     setName("");
  //     setDesc("");
  //     setImage(null);
  //     setIsPublic(false);
  //     loadAddresses();
  //   } catch (error) {
  //     console.error("Erreur lors de l‚Äôajout :", error);
  //     Alert.alert("Erreur", "Impossible d‚Äôajouter l‚Äôadresse.");
  //   }
  // };

  const [loading, setLoading] = useState(false);

const saveAddress = async () => {
  if (!name.trim()) {
    Alert.alert("Erreur", "Veuillez entrer un nom d‚Äôadresse.");
    return;
  }

  setLoading(true);

  try {
    let imageUrl = "";

    if (image) {
      const response = await fetch(image);
      const blob = await response.blob();
      const filename = `images/${Date.now()}_${auth.currentUser?.uid}.jpg`;
      const storageRef = ref(storage, filename);
      await uploadBytes(storageRef, blob);
      imageUrl = await getDownloadURL(storageRef);
    }

    const userEmail = auth.currentUser?.email?.toLowerCase();
    await addDoc(collection(db, "addresses"), {
      name,
      description: desc,
      latitude: selectedCoord.latitude,
      longitude: selectedCoord.longitude,
      imageUrl,
      user: userEmail,
      isPublic: !!isPublic,
      createdAt: new Date(),
    });

    Alert.alert("‚úÖ Succ√®s", "Adresse ajout√©e !");
    setModalVisible(false);
  } catch (error) {
    console.error("Erreur lors de l‚Äôajout :", error);
    Alert.alert("Erreur", "Impossible d‚Äôajouter l‚Äôadresse.");
  } finally {
    setLoading(false);
  }
};


  // üîπ D√©connexion
  const handleLogout = async () => {
    await signOut(auth);
    router.replace("/"); // ‚úÖ renvoie vers la page de connexion
  };

  if (!region)
    return <Text style={{ textAlign: "center", marginTop: 50 }}>Chargement de la carte...</Text>;

  return (
    <View style={styles.container}>
      {/* üåç Carte */}
      <MapView style={styles.map} region={region} onPress={handleAddMarker} onLongPress={handleLongPress}>
        {params?.latitude && params?.longitude && (
          <Marker
            identifier="focused-address"
            coordinate={{
              latitude: parseFloat(String(params.latitude)),
              longitude: parseFloat(String(params.longitude)),
            }}
          >
            {params?.imageUrl ? (
              <Image source={{ uri: String(params.imageUrl) }} style={styles.markerImage} />
            ) : (
              <Image source={require("../../assets/images/icon.png")} style={styles.markerImage} />
            )}
            <Callout>
              <View style={{ maxWidth: 180 }}>
                {params?.imageUrl ? (
                  <Image source={{ uri: String(params.imageUrl) }} style={styles.calloutImage} />
                ) : null}
                {params?.name ? (
                  <Text style={{ fontWeight: "bold", marginTop: 6 }}>{String(params.name)}</Text>
                ) : null}
                {params?.description ? <Text>{String(params.description)}</Text> : null}
              </View>
            </Callout>
          </Marker>
        )}
        {markers.map((marker) => (
          <Marker
            key={marker.id}
            coordinate={{
              latitude: Number(marker.latitude),
              longitude: Number(marker.longitude),
            }}
          >
            {/* üì∏ Si image ‚Üí photo √† la place du pin */}
            {marker.imageUrl ? (
              <Image
                source={{ uri: marker.imageUrl }}
                style={styles.markerImage}
              />
            ) : (
              <Image
                source={require("../../assets/images/icon.png")}
                style={styles.markerImage}
              />
            )}

            {/* üè∑Ô∏è Info-bulle au clic */}
            <Callout>
              <View style={{ maxWidth: 150 }}>
                <Text style={{ fontWeight: "bold" }}>{marker.name}</Text>
                <Text>{marker.description}</Text>
                <Text style={{ fontStyle: "italic", color: "#555" }}>
                  {marker.isPublic ? "üåç Publique" : "üîí Priv√©e"}
                </Text>
              </View>
            </Callout>
          </Marker>
        ))}
      </MapView>

      {/* üî¥ Bouton D√©connexion */}
      <TouchableOpacity style={styles.logoutBtn} onPress={handleLogout}>
        <Text style={styles.logoutText}>D√©connexion</Text>
      </TouchableOpacity>

      {/* üè† Modale d‚Äôajout */}
      <Modal visible={modalVisible} transparent animationType="slide">
        <View style={styles.modalContainer}>
          <View style={styles.modalContent}>
            <Text style={styles.title}>Nouvelle adresse</Text>

            <TextInput
              placeholder="Nom de l‚Äôadresse"
              placeholderTextColor="#999"
              value={name}
              onChangeText={setName}
              style={styles.input}
            />

            <TextInput
              placeholder="Description"
              placeholderTextColor="#999"
              value={desc}
              onChangeText={setDesc}
              style={styles.input}
            />

            <View style={{ flexDirection: "row", alignItems: "center", marginBottom: 10 }}>
              <Text>Adresse publique ? </Text>
              <Switch value={isPublic} onValueChange={setIsPublic} />
            </View>

            <Button title="Choisir une photo" onPress={pickImage} />
            {image && (
              <View style={{ alignItems: "center", marginVertical: 8 }}>
                <Image source={{ uri: image }} style={styles.preview} />
                <Text>üì∏ Image s√©lectionn√©e</Text>
              </View>
            )}

            <Button title={loading ? "Enregistrement..." : "Enregistrer"} disabled={loading} onPress={saveAddress} />
            <Button title="Annuler" onPress={() => setModalVisible(false)} color="red" />
          </View>
        </View>
      </Modal>
    </View>
  );
}

// üíÖ Styles
const styles = StyleSheet.create({
  container: { flex: 1 },
  map: { flex: 1 },
  markerImage: {
    width: 40,
    height: 40,
    borderRadius: 20,
